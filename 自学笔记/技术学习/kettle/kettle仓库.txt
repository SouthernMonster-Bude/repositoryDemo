在pom中添加下列仓库可以拉去kettle pento相关jar包

<repositories>
    <repository>
        <snapshots>
            <enabled>true</enabled>
            <updatePolicy>daily</updatePolicy>
        </snapshots>
        <id>pentaho</id>
        <url>http://nexus.pentaho.org/content/groups/omni/</url>
    </repository>
</repositories>
